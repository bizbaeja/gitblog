# global-style setting

### src/styled.d.ts

- styled.d. ts : **global** 에 원하는 CSS 를 작성하기 전에 type을 선언한다.
  ```css
  import "styled-components";
  import { theme } from "./theme";
  **type Theme = typeof theme;**
  declare module "styled-components" {
    export interface DefaultTheme {
      textColor: string;
      bgColor: string;
      accentColor: string;
    }
  }
  ```

### src/theme.ts

- theme.ts : 그리고 theme 이라는 CSS 컴포넌트를 만들어준다. 여기서 필요한 속성을 꺼내 사용하면 된다.
  ```css
  import { DefaultTheme } from "styled-components";

  export const theme: DefaultTheme = {
    bgColor: "#2f3640",
    textColor: "#f5f6fa",
    accentColor: "#4cd137",
  };
  ```

- Router.tsx
  ```css
  import { BrowserRouter, Routes, Route } from "react-router-dom";
  import Header from "./component/Header";
  import Coin from "./screens/Coin";
  import Coins from "./screens/Coins";
  function Router() {
    return (
      <BrowserRouter>
        <Header />
        <Routes>
          <Route path="/" element={<Coin />}></Route>
          <Route path="/coins" element={<Coins />}></Route>
        </Routes>
      </BrowserRouter>
    );
  }
  export default Router;
  ```

### src/index.tsx

- index.tsx
  ```css
  import React from "react";
  import ReactDOM from "react-dom/client";
  import App from "./App";
  //커스텀한 CSS 컴포넌트 import 해오기
  import { ThemeProvider } from "styled-components";
  // css 컴포넌트 임포트 해오기
  import { theme } from "./theme";
  import GlobalStyles from "./global";
  const root = ReactDOM.createRoot(
    document.getElementById("root") as HTMLElement
  );

  root.render(
  //themeprovider 안에 theme 적용
    <ThemeProvider theme={theme}>
  // 적용하고자 하는 컴포넌트의 형제요소로 GlobalStyles 컴포넌트를 위치시킨다.
      <GlobalStyles />
      <React.StrictMode>
        <App></App>
      </React.StrictMode>
    </ThemeProvider>
  );
  ```
